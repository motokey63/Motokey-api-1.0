function startEditDevis(id) {
  const dv = devisListCache.find(x=>x.id===id);
  if (!dv) { toast('❌ Devis introuvable'); return; }
  devisEditId = id;
  devisLines = (dv.lignes && dv.lignes.length) ? JSON.parse(JSON.stringify(dv.lignes)) : [{designation:'',qte:1,prix_ht:0}];
  const zone = document.getElementById('devisFormZone');
  if (zone) zone.innerHTML = renderDevisFormCard();
}

function cancelEditDevis() {
  devisEditId = null;
  devisLines = [{designation:'',qte:1,prix_ht:0}];
  const zone = document.getElementById('devisFormZone');
  if (zone) zone.innerHTML = renderDevisFormCard();
}

async function saveDevis() {

  const lignes = devisLines.filter(l=>l.designation.trim());

}
