```typescript
export interface SessionRefresher {
  getValidAccessToken(): Promise<string | null>; // returns current token if >60s left, else single-flight refresh; null on hard expiry
  refreshIfNeeded(): Promise<boolean>;            // proactive: refresh when <REFRESH_SKEW_SECS left; true if session still valid
}
export function createSessionRefresher(deps: {
  getSession: () => AuthSession | null;
  setSession: (s: AuthSession) => Promise<void>;   // persist + update state
  onHardExpiry: () => Promise<void>;               // clear + set unauthenticated + toast once (D-08)
}): SessionRefresher;
```

    - getValidAccessToken with a token that has <60s left triggers exactly ONE /auth/client/refresh call and returns the new access token
    - 5 concurrent getValidAccessToken calls while the token is expiring produce exactly ONE underlying refresh call (single-flight) — all 5 resolve to the same new token
    - refreshIfNeeded with a token >300s left returns true and does NOT call refresh
    - refreshIfNeeded with a token <300s left calls refresh once and returns true on success
    - When the refresh call returns ok:false, onHardExpiry is invoked once and getValidAccessToken resolves to null
    - After a completed refresh, a later call with the (new) still-valid token does not re-refresh (the in-flight promise is cleared)
